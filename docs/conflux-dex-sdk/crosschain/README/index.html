#Conflux Crosschain SDK

#Functions

// get data payloads for ethereum mint transaction
// token: token type, ('eth', 'usdt')
// toAddress: conflux target address
// amount: Number | String
// return type: string
function ethMint(token, toAddress, amount);


// get op_return data for bitcoin mint transaction
// toAddress: conflux target address
// return type: string
function btcMint(toAddress);


// send a conflux transaction to burn specific token.
// cfx: conflux node provider
// sender: conflux wallet object
// contractAddress: conflux crosschain contract address
// token: token type, ('eth', 'usdt')
// toAddress: ethereum target address
// amount: Number | String
// return type: Promise
async function ethBurn(cfx, sender, contractAddress, token, toAddress, amount);


// send a conflux transaction to burn cBTC.
// cfx: conflux node provider
// sender: conflux wallet object
// contractAddress: conflux crosschain contract address
// toAddress: ethereum target address
// amount: Number | String
// isTest: toAddress is bitcoin testnet3 address (=true) or mainnet address (=false)
// return type: Promise
async function btcBurn(cfx, sender, contractAddress, toAddress, amount, isTest = false);


// get user mint/burn operation list.
// data: JSON object
// {
//    address: (String) user conflux address,
//    types: (array of String) subset of {"eth_mint", "eth_burn", "erc20_mint", "erc20_burn", "btc_mint", "btc_burn"},
//    status: (array of String) subset of {"doing", "finished"},
// }
// skip: the number of leading results to be ignored
// cnt: the maximum number of results to return
// node_url: url of running crosschain custodian node
async function getUserOperationList(data, skip, cnt, node_url);


// get ethereum receive wallet address (used for both eth & usdt) of user.
// user_address: conflux address of user
// node_url: url of running crosschain custodian node
async function getUserReceiveWalletEth(user_address, node_url);


// get bitcoin receive wallet address (used for both eth & usdt) of user.
// user_address: conflux address of user
// node_url: url of running crosschain custodian node
async function getUserReceiveWalletBtc(user_address, node_url);

#Examples

Import crosschain module.

const crosschain = require('crosschain');

Get mint payload for eth and send transaction using web3js.

const crosschain = require('conflux-crosschain');
const ethpub = /* your ethereum address */;
const ethpri = /* your ethereum private key */;
const toAddress = /* target conflux address */;
const test_conf = crosschain.test_conf;


// initialize node connection
const Web3 = require('web3');
const provider_url = /* web3js provider */
const w3 = new Web3(provider_url)


async function mintEth() {
  let amount = 1000000000;
  let nonce = await w3.eth.getTransactionCount(ethpub);
  const txParams = {
    from: ethpub,
    to: test_conf.eth_factory_addr,
    nonce: w3.utils.toHex(nonce),
    gas: w3.utils.toHex(2000000),
    gasPrice: "2000000000",
    value: w3.utils.toHex(amount),
    data: crosschain.ethMint('eth', toAddress, amount),
  };
  const {rawTransaction} = await w3.eth.accounts.signTransaction(txParams, ethpri);
  const receipt = await w3.eth.sendSignedTransaction(rawTransaction);
  return receipt;
}

Mint of usdt is quite similar as eth, but before mint, user need to give eth_factory contract proper allowance:

let amount = 100000;


async function approve(amount) {
  let nonce = await w3.eth.getTransactionCount(ethpub);
  const txParams = {
    from: ethpub,
    to: test_conf.usdt_addr,
    nonce: w3.utils.toHex(nonce),
    gas: w3.utils.toHex(2000000),
    gasPrice: "2000000000",
    data: crosschain.erc20Approve(test_conf.eth_factory_addr, amount),
  };
  const {rawTransaction} = await w3.eth.accounts.signTransaction(txParams, ethpri);
  const receipt = await w3.eth.sendSignedTransaction(rawTransaction);
  console.log(123);
  if (!receipt.status)
    process.exit(1);
}


async function mint() {
  let nonce = await w3.eth.getTransactionCount(ethpub);
